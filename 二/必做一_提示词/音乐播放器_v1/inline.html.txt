// 播放器状态
let isPlaying = false;
let currentTime = 0;
let totalTime = 210; // 3:30
let volume = 70;

// DOM元素
const playStatus = document.getElementById('playStatus');
const playBtn = document.getElementById('playBtn');
const prevBtn = document.getElementById('prevBtn');
const nextBtn = document.getElementById('nextBtn');
const progress = document.getElementById('progress');
const progressHandle = document.getElementById('progressHandle');
const progressBar = document.querySelector('.progress-bar');
const currentTimeEl = document.getElementById('currentTime');
const totalTimeEl = document.getElementById('totalTime');
const volumeEl = document.getElementById('volume');
const volumeHandle = document.getElementById('volumeHandle');
const volumeBar = document.querySelector('.volume-bar');

// 格式化时间
function formatTime(seconds) {
    const mins = Math.floor(seconds / 60);
    const secs = Math.floor(seconds % 60);
    return `${mins}:${secs.toString().padStart(2, '0')}`;
}

// 更新进度条
function updateProgress() {
    const percent = (currentTime / totalTime) * 100;
    progress.style.width = `${percent}%`;
    progressHandle.style.left = `${percent}%`;
    currentTimeEl.textContent = formatTime(currentTime);
}

// 更新音量条
function updateVolume() {
    volumeEl.style.width = `${volume}%`;
    volumeHandle.style.left = `${volume}%`;
}

// 播放/暂停切换
function togglePlay() {
    isPlaying = !isPlaying;
    playStatus.textContent = isPlaying ? '正在播放' : '播放暂停';
    
    if (isPlaying) {
        // 模拟播放进度
        playInterval = setInterval(() => {
            if (currentTime < totalTime) {
                currentTime += 1;
                updateProgress();
            } else {
                clearInterval(playInterval);
                isPlaying = false;
                playStatus.textContent = '播放暂停';
                currentTime = 0;
                updateProgress();
            }
        }, 1000);
    } else {
        clearInterval(playInterval);
    }
}

// 上一首
function playPrev() {
    // 模拟上一首
    currentTime = 0;
    updateProgress();
    if (isPlaying) {
        clearInterval(playInterval);
        togglePlay();
        togglePlay();
    }
}

// 下一首
function playNext() {
    // 模拟下一首
    currentTime = 0;
    updateProgress();
    if (isPlaying) {
        clearInterval(playInterval);
        togglePlay();
        togglePlay();
    }
}

// 进度条拖拽
let isDraggingProgress = false;

function startDragProgress(e) {
    isDraggingProgress = true;
    updateProgressFromMouse(e);
}

function updateProgressFromMouse(e) {
    if (!isDraggingProgress) return;
    
    const rect = progressBar.getBoundingClientRect();
    const offsetX = e.clientX - rect.left;
    const percent = Math.max(0, Math.min(100, (offsetX / rect.width) * 100));
    
    currentTime = (percent / 100) * totalTime;
    updateProgress();
}

function stopDragProgress() {
    isDraggingProgress = false;
}

// 音量条拖拽
let isDraggingVolume = false;

function startDragVolume(e) {
    isDraggingVolume = true;
    updateVolumeFromMouse(e);
}

function updateVolumeFromMouse(e) {
    if (!isDraggingVolume) return;
    
    const rect = volumeBar.getBoundingClientRect();
    const offsetX = e.clientX - rect.left;
    volume = Math.max(0, Math.min(100, (offsetX / rect.width) * 100));
    updateVolume();
}

function stopDragVolume() {
    isDraggingVolume = false;
}

// 点击进度条
function clickProgressBar(e) {
    const rect = progressBar.getBoundingClientRect();
    const offsetX = e.clientX - rect.left;
    const percent = Math.max(0, Math.min(100, (offsetX / rect.width) * 100));
    
    currentTime = (percent / 100) * totalTime;
    updateProgress();
}

// 点击音量条
function clickVolumeBar(e) {
    const rect = volumeBar.getBoundingClientRect();
    const offsetX = e.clientX - rect.left;
    volume = Math.max(0, Math.min(100, (offsetX / rect.width) * 100));
    updateVolume();
}

// 事件监听
playBtn.addEventListener('click', togglePlay);
prevBtn.addEventListener('click', playPrev);
nextBtn.addEventListener('click', playNext);

// 进度条事件
progressBar.addEventListener('click', clickProgressBar);
progressHandle.addEventListener('mousedown', startDragProgress);
document.addEventListener('mousemove', updateProgressFromMouse);
document.addEventListener('mouseup', stopDragProgress);

// 音量条事件
volumeBar.addEventListener('click', clickVolumeBar);
volumeHandle.addEventListener('mousedown', startDragVolume);
document.addEventListener('mousemove', updateVolumeFromMouse);
document.addEventListener('mouseup', stopDragVolume);

// 初始化
function init() {
    totalTimeEl.textContent = formatTime(totalTime);
    updateProgress();
    updateVolume();
}

init();
